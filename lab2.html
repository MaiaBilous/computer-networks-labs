Лабораторна робота №2

Завдання

Створити програму Python для перевірки доступності не тільки одного хоста, а й діапазону IP-адрес.

Принцип роботи

Користувач вводить одну IP-адресу або діапазон у форматі startIP-endIP. Програма послідовно виконує ping для кожної адреси та виводить результат.

Код програми


from pythonping import ping
import ipaddress

def ping_host(host, count):
    print(f"\nPing {host}")
    try:
        result = ping(host, count=count)
        print(result)
    except Exception as err:
        print(f"Помилка: {err}")

def ip_range(start_ip, end_ip):
    start = ipaddress.IPv4Address(start_ip)
    end = ipaddress.IPv4Address(end_ip)

    if start > end:
        start, end = end, start

    current = start
    while current <= end:
        yield str(current)
        current += 1

def main():
    host = input("Введіть адресу або діапазон: ")
    count = int(input("Кількість пакетів: "))

    if "-" in host:
        start_ip, end_ip = host.split("-")
        for ip in ip_range(start_ip, end_ip):
            ping_host(ip, count)
    else:
        ping_host(host, count)

if __name__ == "__main__":
    main()
