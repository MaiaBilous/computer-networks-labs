Лабораторна робота №3

Завдання

Розширити програму з лабораторної роботи №2, додавши:

  • Сортування відповідей від пінгу (працюють / не працюють)
  • Збереження робочих сайтів у файл working_hosts.txt
  • Збереження неробочих сайтів у файл dead_hosts.txt

Принцип роботи програми

Програма перевіряє доступність кожного хоста з діапазону IP-адрес. Після завершення перевірки всі відповіді сортуються на дві групи:

  • Працюючі хости – ті, які відповіли хоча б на один пакет
  • Непрацюючі хости – ті, які не відповіли або видали помилку

Результати автоматично зберігаються у відповідні текстові файли та виводяться на екран у відсортованому вигляді.

Код програми

ping_sorted.py


from pythonping import ping
import ipaddress
import sys

# Файли для збереження результатів
WORKING_FILE = "working_hosts.txt"
DEAD_FILE = "dead_hosts.txt"

def check_host(host, count=2):
    """
    Перевіряє доступність одного хоста.
    Повертає (host, status, details)
    status: True - працює, False - не працює
    """
    try:
        result = ping(host, count=count, timeout=2)
        # Якщо є хоча б одна відповідь - хост працює
        if result.success():
            return (host, True, str(result))
        else:
            return (host, False, "Немає відповіді")
    except Exception as err:
        return (host, False, f"Помилка: {err}")

def ip_range(start_ip, end_ip):
    """Генератор IP-адрес у діапазоні"""
    start = ipaddress.IPv4Address(start_ip)
    end = ipaddress.IPv4Address(end_ip)
    if start > end:
        start, end = end, start
    current = start
    while current <= end:
        yield str(current)
        current += 1

def save_results(working, dead):
    """Зберігає результати у файли"""
    # Збереження робочих хостів
    with open(WORKING_FILE, "w", encoding="utf-8") as f:
        f.write("ПРАЦЮЮЧІ ХОСТИ:\n")
        f.write("=" * 40 + "\n")
        for host, details in working:
            f.write(f"{host}\n")
            f.write(f"  {details}\n\n")
    
    # Збереження неробочих хостів
    with open(DEAD_FILE, "w", encoding="utf-8") as f:
        f.write("НЕПРАЦЮЮЧІ ХОСТИ:\n")
        f.write("=" * 40 + "\n")
        for host, details in dead:
            f.write(f"{host}\n")
            f.write(f"  {details}\n\n")
    
    print(f"\n Результати збережено:")
    print(f"   Працюючі: {WORKING_FILE}")
    print(f"   Непрацюючі: {DEAD_FILE}")

def main():
    print("=" * 50)
    print("     ПЕРЕВІРКА ДОСТУПНОСТІ ХОСТІВ")
    print("=" * 50)
    
    host_input = input("\nВведіть адресу або діапазон (напр. 192.168.1.1 або 192.168.1.1-192.168.1.10): ")
    count = int(input("Кількість пакетів на хост: "))
    
    # Збираємо всі адреси для перевірки
    hosts_to_check = []
    if "-" in host_input:
        start_ip, end_ip = host_input.split("-")
        hosts_to_check = list(ip_range(start_ip.strip(), end_ip.strip()))
    else:
        hosts_to_check = [host_input.strip()]
    
    print(f"\n🔍 Перевіряємо {len(hosts_to_check)} хост(ів)...\n")
    
    working_hosts = []  # (host, details)
    dead_hosts = []     # (host, details)
    
    for i, host in enumerate(hosts_to_check, 1):
        print(f"[{i}/{len(hosts_to_check)}] Перевірка {host}...", end=" ")
        hostname, status, details = check_host(host, count)
        
        if status:
            print(" ПРАЦЮЄ")
            working_hosts.append((hostname, details))
        else:
            print(" НЕ ПРАЦЮЄ")
            dead_hosts.append((hostname, details))
    
    # Виведення відсортованих результатів
    print("\n" + "=" * 50)
    print("     ВІДСОРТОВАНІ РЕЗУЛЬТАТИ")
    print("=" * 50)
    
    print(f"\n ПРАЦЮЮЧІ ХОСТИ ({len(working_hosts)}):")
    for host, details in working_hosts:
        print(f"   • {host}")
    
    print(f"\n НЕПРАЦЮЮЧІ ХОСТИ ({len(dead_hosts)}):")
    for host, details in dead_hosts:
        print(f"   • {host}")
    
    # Збереження у файли
    save_results(working_hosts, dead_hosts)
    
    print("\n🏁 Роботу завершено.")

if __name__ == "__main__":
    main()
