Лабораторна робота №4

Завдання

Додати можливості фільтрації моніторингу в реальному часі за:

  • Зовнішніми або внутрішніми IP-адресами
  • Протоколами (TCP, UDP, ICMP)
  • Портами

Принцип роботи програми

Програма виконує моніторинг мережевих вузлів у реальному часі та дозволяє динамічно фільтрувати результати за різними критеріями.

Користувач може:

  • Фільтрувати лише зовнішні (публічні) або лише внутрішні (приватні) IP-адреси
  • Вказати протокол для перевірки (ICMP для ping, або TCP для портів)
  • Вказати порт для перевірки (при використанні TCP)
Примітка: Для перевірки TCP-портів використовується бібліотека socket. Для роботи з ICMP (ping) використовується pythonping.

Код програми

monitor_filter.py


from pythonping import ping
import ipaddress
import socket
import sys
import time

def is_private_ip(ip):
    """Перевіряє, чи є IP-адреса внутрішньою (приватною)"""
    try:
        ip_obj = ipaddress.ip_address(ip)
        return ip_obj.is_private
    except:
        return False

def is_public_ip(ip):
    """Перевіряє, чи є IP-адреса зовнішньою (публічною)"""
    return not is_private_ip(ip)

def check_ping(host, count=2, timeout=2):
    """Перевірка доступності через ICMP (ping)"""
    try:
        result = ping(host, count=count, timeout=timeout)
        return result.success()
    except:
        return False

def check_tcp_port(host, port, timeout=2):
    """Перевірка доступності TCP-порту"""
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        result = sock.connect_ex((host, port))
        sock.close()
        return result == 0
    except:
        return False

def get_ip_filter_choice():
    """Отримує вибір фільтрації за типом IP"""
    print("\nФільтрація за IP-адресами:")
    print("  1. Всі адреси")
    print("  2. Тільки внутрішні (приватні) IP")
    print("  3. Тільки зовнішні (публічні) IP")
    choice = input("Ваш вибір (1-3): ").strip()
    return choice

def get_protocol_choice():
    """Отримує вибір протоколу"""
    print("\nВиберіть протокол:")
    print("  1. ICMP (ping)")
    print("  2. TCP (перевірка порту)")
    choice = input("Ваш вибір (1-2): ").strip()
    return choice

def apply_ip_filter(host, filter_type):
    """
    Застосовує фільтр до окремого хоста.
    Повертає True, якщо хост проходить фільтр.
    """
    if filter_type == "1":  # Всі адреси
        return True
    elif filter_type == "2":  # Тільки внутрішні
        return is_private_ip(host)
    elif filter_type == "3":  # Тільки зовнішні
        return is_public_ip(host)
    return True

def monitor_hosts(hosts, protocol, port, ip_filter, count=2):
    """
    Моніторинг хостів у реальному часі з фільтрацією.
    """
    print("\n" + "=" * 60)
    print("     МОНІТОРИНГ У РЕАЛЬНОМУ ЧАСІ")
    print("=" * 60)
    
    protocol_name = "ICMP (ping)" if protocol == "1" else f"TCP (порт {port})"
    ip_filter_name = {
        "1": "Всі адреси",
        "2": "Тільки внутрішні (приватні)",
        "3": "Тільки зовнішні (публічні)"
    }.get(ip_filter, "Всі адреси")
    
    print(f"\n Налаштування фільтрації:")
    print(f"   • IP-фільтр: {ip_filter_name}")
    print(f"   • Протокол: {protocol_name}")
    print(f"   • Хостів у списку: {len(hosts)}")
    
    # Фільтруємо хости за IP-типом
    filtered_hosts = [h for h in hosts if apply_ip_filter(h, ip_filter)]
    print(f"   • Після IP-фільтра: {len(filtered_hosts)}")
    
    print("\n Починаємо моніторинг... (Ctrl+C для виходу)\n")
    
    try:
        round_num = 1
        while True:
            print(f"\n--- Цикл моніторингу #{round_num} ---")
            print(f"Час: {time.strftime('%H:%M:%S')}")
            
            working = []
            dead = []
            
            for host in filtered_hosts:
                if protocol == "1":  # ICMP
                    status = check_ping(host, count)
                else:  # TCP
                    status = check_tcp_port(host, port)
                
                if status:
                    working.append(host)
                    print(f"   {host} - ДОСТУПНИЙ")
                else:
                    dead.append(host)
                    print(f"   {host} - НЕДОСТУПНИЙ")
            
            print(f"\n Статистика циклу:")
            print(f"   Доступні: {len(working)}")
            print(f"   Недоступні: {len(dead)}")
            
            round_num += 1
            time.sleep(5)  # Пауза між циклами
    except KeyboardInterrupt:
        print("\n\n Моніторинг зупинено користувачем.")

def main():
    print("=" * 60)
    print("     МОНІТОРИНГ З ФІЛЬТРАЦІЄЮ В РЕАЛЬНОМУ ЧАСІ")
    print("=" * 60)
    
    # Введення списку хостів
    print("\nВведіть IP-адреси для моніторингу (через кому або пробіл)")
    print("Приклад: 8.8.8.8, 192.168.1.1, 1.1.1.1, 10.0.0.1")
    hosts_input = input("IP-адреси: ")
    
    # Розбираємо введені адреси
    hosts = []
    for part in hosts_input.replace(",", " ").split():
        if part.strip():
            hosts.append(part.strip())
    
    if not hosts:
        print(" Не вказано жодної IP-адреси!")
        return
    
    # Вибір фільтрації за типом IP
    ip_filter = get_ip_filter_choice()
    
    # Вибір протоколу
    protocol = get_protocol_choice()
    
    port = None
    if protocol == "2":  # TCP
        port = int(input("Введіть номер порту для перевірки: "))
    
    # Запуск моніторингу
    monitor_hosts(hosts, protocol, port, ip_filter)

if __name__ == "__main__":
    main()

Пояснення до коду:
  • is_private_ip() – визначає, чи є адреса внутрішньою (приватною), використовуючи модуль ipaddress
  • is_public_ip() – перевіряє, чи є адреса зовнішньою (публічною)
  • check_ping() – перевіряє доступність через ICMP-запити
  • check_tcp_port() – перевіряє доступність TCP-порту через socket
  • monitor_hosts() – виконує моніторинг у нескінченному циклі з фільтрацією та виведенням статистики
